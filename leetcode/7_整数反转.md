class Solution {
    public int reverse(int x) {
        int maxTen = Integer.MAX_VALUE / 10, minTen = Integer.MIN_VALUE / 10;
        int ans = 0;

        while (x != 0) {
            int tmp = x % 10;
            if ((ans > maxTen || (ans == maxTen && tmp > 7))
                    || (ans < minTen || (ans == minTen && tmp < -8))) {
                return 0;
            }
            ans = ans * 10 + tmp;
            x /= 10;
        }

        return ans;
    }
}
